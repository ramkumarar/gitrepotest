local wezterm = require("wezterm")
local config = wezterm.config_builder()

-- 1. Default shell profile (Portable Git Bash)
config.default_prog = { "C:\\dev\\apps\\PortableGit\\bin\\bash.exe", "--login", "-i" }

-- 2. Theme 
config.color_scheme = "Rosé Pine (official)"

-- 3. Absolute Bypass for Close Confirmations (FIXED)
config.window_close_confirmation = "NeverPrompt"

-- Windows systems track the absolute executable file extensions (.exe). 
-- This list catches every variant of your custom paths so it never triggers prompts.
config.skip_close_confirmation_for_processes_named = { 
  "bash", 
  "bash.exe", 
  "cmd", 
  "cmd.exe", 
  "conhost.exe" 
}

-- 4. Enable both Git Bash and Windows CMD in the Launcher Menu
config.launch_menu = {
  {
    label = "Portable Git Bash",
    args = { "C:\\dev\\apps\\PortableGit\\bin\\bash.exe", "--login", "-i" },
  },
  {
    label = "Windows Command Prompt (CMD)",
    args = { "cmd.exe" },
  },
}

-- 5. Keybindings for Multi-Shells & Forceful Closing (FIXED)
config.keys = {
  -- Paste via CTRL + V
  { key = "v", mods = "CTRL", action = wezterm.action.PasteFrom("Clipboard") },
  -- Copy via CTRL + C
  { key = "c", mods = "CTRL", action = wezterm.action.CopyTo("Clipboard") },

  -- Force close current tab instantly via hotkey without checking rules (confirm = false)
  {
    key = "w",
    mods = "CTRL|SHIFT",
    action = wezterm.action.CloseCurrentTab { confirm = false },
  },

  -- Force close current pane instantly via hotkey (confirm = false)
  {
    key = "x",
    mods = "CTRL|SHIFT",
    action = wezterm.action.CloseCurrentPane { confirm = false },
  },

  -- Press CTRL + SHIFT + L to quickly open the profile selector menu
  {
    key = "l",
    mods = "CTRL|SHIFT",
    action = wezterm.action.ShowLauncher,
  },
}

return config
